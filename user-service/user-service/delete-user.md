@Override
public void deleteUser(UUID id) {
    validateNotNull(id);
    for (var i = 0; i<users.size(); i++) {
        if (users.get(i).getId().equals(id)) {
            users.remove(i);
            return;
        }
    }
    throw new NoSuchElementException();
}
